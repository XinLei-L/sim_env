197 inline utils::Vector2d SocialForceModel::computeDesiredForce(Agent& agent) const
198 {
199  utils::Vector2d desiredDirection;
200  if (!agent.goals.empty() && (agent.goals.front().center - agent.position).norm() > agent.goals.front().radius)
201  {
202  utils::Vector2d diff = agent.goals.front().center - agent.position;
203  desiredDirection = diff.normalized();
204  agent.forces.desiredForce = agent.params.forceFactorDesired *
205  (desiredDirection * agent.desiredVelocity - agent.velocity) /
206  agent.params.relaxationTime;
207  agent.antimove = false;
208  }
209  else
210  {
211  agent.forces.desiredForce = -agent.velocity / agent.params.relaxationTime;
212  agent.antimove = true;
213  }
214  return desiredDirection;
215 }

272 inline void SocialForceModel::computeSocialForce(unsigned index, std::vector<Agent>& agents) const
273 {
274  Agent& agent = agents[index];
275  agent.forces.socialForce.set(0, 0);
276  for (unsigned i = 0; i < agents.size(); i++)
277  {
278  if (i == index)
279  {
280  continue;
281  }
282  utils::Vector2d diff = agents[i].position - agent.position;
283  utils::Vector2d diffDirection = diff.normalized();
284  utils::Vector2d velDiff = agent.velocity - agents[i].velocity;
285  utils::Vector2d interactionVector = agent.params.lambda * velDiff + diffDirection;
286  double interactionLength = interactionVector.norm();
287  utils::Vector2d interactionDirection = interactionVector / interactionLength;
288  utils::Angle theta = interactionDirection.angleTo(diffDirection);
289  double B = agent.params.gamma * interactionLength;
290  double thetaRad = theta.toRadian();
291  double forceVelocityAmount = -std::exp(-diff.norm() / B - PW(agent.params.nPrime * B * thetaRad));
292  double forceAngleAmount = -theta.sign() * std::exp(-diff.norm() / B - PW(agent.params.n * B * thetaRad));
293  utils::Vector2d forceVelocity = forceVelocityAmount * interactionDirection;
294  utils::Vector2d forceAngle = forceAngleAmount * interactionDirection.leftNormalVector();
295  agent.forces.socialForce += agent.params.forceFactorSocial * (forceVelocity + forceAngle);
296  if (i == 0)
297  {
298  agent.forces.robotSocialForce = agent.params.forceFactorSocial * (forceVelocity + forceAngle);
299  }

335 inline void SocialForceModel::computeGroupForce(unsigned index, const utils::Vector2d& desiredDirection,
336  std::vector<Agent>& agents,
337  const std::unordered_map<int, Group>& groups) const
338 {
339  Agent& agent = agents[index];
340  agent.forces.groupForce.set(0, 0);
341  agent.forces.groupGazeForce.set(0, 0);
342  agent.forces.groupCoherenceForce.set(0, 0);
343  agent.forces.groupRepulsionForce.set(0, 0);
344  if (groups.count(agent.groupId) == 0 || groups.at(agent.groupId).agents.size() < 2)
345  {
346  return;
347  }
348  const Group& group = groups.at(agent.groupId);
349 
350  // Gaze force
351  utils::Vector2d com = group.center;
352  com = (1 / (double)(group.agents.size() - 1)) * (group.agents.size() * com - agent.position);
353 
354  utils::Vector2d relativeCom = com - agent.position;
355  utils::Angle visionAngle = utils::Angle::fromDegree(90);
356  double elementProduct = desiredDirection.dot(relativeCom);
357  utils::Angle comAngle =
358  utils::Angle::fromRadian(std::acos(elementProduct / (desiredDirection.norm() * relativeCom.norm())));
359  if (comAngle > visionAngle)
360  {
361 #ifdef _PAPER_VERSION_
362  utils::Angle necessaryRotation = comAngle - visionAngle;
363  agent.forces.groupGazeForce = -necessaryRotation.toRadian() * desiredDirection;
364 #else
365  double desiredDirectionSquared = desiredDirection.squaredNorm();
366  double desiredDirectionDistance = elementProduct / desiredDirectionSquared;
367  agent.forces.groupGazeForce = desiredDirectionDistance * desiredDirection;
368 #endif
369  agent.forces.groupGazeForce *= agent.params.forceFactorGroupGaze;
370  }
371 
372  // Coherence force
373  com = group.center;
374  relativeCom = com - agent.position;
375  double distance = relativeCom.norm();
376  double maxDistance = ((double)group.agents.size() - 1) / 2;
377 #ifdef _PAPER_VERSION_
378  if (distance >= maxDistance)
379  {
380  agent.forces.groupCoherenceForce = relativeCom.normalized();
381  agent.forces.groupCoherenceForce *= agent.params.forceFactorGroupCoherence;
382  }
383 #else
384  agent.forces.groupCoherenceForce = relativeCom;
385  double softenedFactor = agent.params.forceFactorGroupCoherence * (std::tanh(distance - maxDistance) + 1) / 2;
386  agent.forces.groupCoherenceForce *= softenedFactor;
387 #endif
388 
389  // Repulsion Force
390  for (unsigned i = 0; i < group.agents.size(); i++)
391  {
392  if (index == group.agents[i])
393  {
394  continue;
395  }
396  utils::Vector2d diff = agent.position - agents.at(group.agents[i]).position;
397  if (diff.norm() < agent.radius + agents.at(group.agents[i]).radius)
398  {
399  agent.forces.groupRepulsionForce += diff;
400  }
401  }
402  agent.forces.groupRepulsionForce *= agent.params.forceFactorGroupRepulsion;
403 
404  // Group Force
405  agent.forces.groupForce =
406  agent.forces.groupGazeForce + agent.forces.groupCoherenceForce + agent.forces.groupRepulsionForce;
407 }

478 {

484  continue;

501  agents[i].forces.obstacleForce + agents[i].forces.groupForce;

548 inline std::vector<Agent>& SocialForceModel::updatePosition(std::vector<Agent>& agents, double dt) const
549 {
550  for (unsigned i = 0; i < agents.size(); i++)
551  {
552  utils::Vector2d initPos = agents[i].position;
553  if (agents[i].teleoperated)
554  {
555  double imd = agents[i].linearVelocity * dt;
556  utils::Vector2d inc(imd * std::cos(agents[i].yaw.toRadian() + agents[i].angularVelocity * dt * 0.5),
557  imd * std::sin(agents[i].yaw.toRadian() + agents[i].angularVelocity * dt * 0.5));
558  agents[i].yaw += utils::Angle::fromRadian(agents[i].angularVelocity * dt);
559  agents[i].position += inc;
560  agents[i].velocity.set(agents[i].linearVelocity * agents[i].yaw.cos(),
561  agents[i].linearVelocity * agents[i].yaw.sin());
562  }
563  else
564  {
565  agents[i].velocity += agents[i].forces.globalForce * dt;
566  if (agents[i].velocity.norm() > agents[i].desiredVelocity)
567  {
568  agents[i].velocity.normalize();
569  agents[i].velocity *= agents[i].desiredVelocity;
570  }
571  agents[i].yaw = agents[i].velocity.angle();
572  agents[i].position += agents[i].velocity * dt;
573  }
574  agents[i].movement = agents[i].position - initPos;
575  if (!agents[i].goals.empty() &&
576  (agents[i].goals.front().center - agents[i].position).norm() <= agents[i].goals.front().radius)
577  {
578  Goal g = agents[i].goals.front();
579  agents[i].goals.pop_front();
580  if (agents[i].cyclicGoals)
581  {
582  agents[i].goals.push_back(g);
583  }
584  }
585  }
586  return agents;
