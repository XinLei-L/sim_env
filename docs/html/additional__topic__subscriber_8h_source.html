89 template<class MessageType>
90 class AdditionalTopicSubscriber: public _AdditionalTopicSubscriber
91 {
92 // No Q_OBJECT macro here, moc does not support Q_OBJECT in a templated class.
93 public:
96  typedef AdditionalTopicSubscriber<MessageType> ATSClass;
97 
98  AdditionalTopicSubscriber(const QString& propertyName, Display* display, DisplayContext* context, ros::NodeHandle& update_nh, const boost::function<void (boost::shared_ptr<const MessageType>)>& messageCallback)
99  : tf_filter(NULL), m_messagesReceived(0), m_display(display), m_context(context), m_updateNodeHandle(update_nh), m_enabled(false), m_messageCallback(messageCallback)
100  {
101  _AdditionalTopicSubscriber::initialize(display, context->getFrameManager());
102 
103  additional_topic_property_->setName(propertyName);
104  QString message_type = QString::fromStdString(ros::message_traits::datatype<MessageType>());
105  additional_topic_property_->setMessageType(message_type);
106  additional_topic_property_->setDescription(message_type + " topic to subscribe to.");
107 
108 #if ROS_VERSION_MINIMUM(1, 14, 0)
109  tf_filter = new tf2_ros::MessageFilter<MessageType>(*m_context->getTF2BufferPtr(), "map", 10, m_updateNodeHandle);
110 #else
111  tf_filter = new tf::MessageFilter<MessageType>(*m_context->getTFClient(), "map", 10, m_updateNodeHandle);
112 #endif
113 
114  tf_filter->connectInput(m_subscriber);
115  tf_filter->registerCallback(boost::bind(&AdditionalTopicSubscriber<MessageType>::incomingMessage, this, _1));
116  m_context->getFrameManager()->registerFilterForTransformStatusCheck(tf_filter, display);
117 
118  setEnabled(m_display->isEnabled());
119  updateTopic();
120  fixedFrameChanged();
121  }
122 
123  virtual ~AdditionalTopicSubscriber()
124  {
125  unsubscribe();
126  delete tf_filter;
127  }
128 
129  virtual void reset()
130  {
131  tf_filter->clear();
132  m_messagesReceived = 0;
133  }
134 
135  void setEnabled(bool enabled)
136  {
137  m_enabled = enabled;
138  if (enabled) subscribe();
139  }
140 
141 protected:
142  virtual void updateTopic()
143  {
144  ROS_DEBUG_STREAM_NAMED("AdditionalTopicSubscriber", "AdditionalTopicSubscriber: Topic was changed to " << additional_topic_property_->getTopicStd());
145  unsubscribe();
146  reset();
147  subscribe();
148  m_context->queueRender();
149  }
150 
151  virtual void displayEnableChanged()
152  {
153  ROS_DEBUG_STREAM_NAMED("AdditionalTopicSubscriber", "AdditionalTopicSubscriber: Display enabled = " << m_display->getBool());
154  setEnabled(m_display->getBool());
155  }
156 
157  virtual void fixedFrameChanged()
158  {
159  ROS_DEBUG_STREAM_NAMED("AdditionalTopicSubscriber", "AdditionalTopicSubscriber: Fixed frame has changed for topic " << additional_topic_property_->getTopicStd());
160  tf_filter->setTargetFrame(m_context->getFixedFrame().toStdString());
161  reset();
162  }
163 
164  virtual void subscribe()
165  {
166  if (!m_display->isEnabled()) {
167  return;
168  }
169 
170  try {
171  ROS_DEBUG_STREAM_NAMED("AdditionalTopicSubscriber", "AdditionalTopicSubscriber: Subscribing to topic " << additional_topic_property_->getTopicStd());
172  m_subscriber.subscribe(m_updateNodeHandle, additional_topic_property_->getTopicStd(), 10);
173  m_display->setStatus(StatusProperty::Ok, additional_topic_property_->getName(), "OK");
174  }
175  catch (ros::Exception& e) {
176  m_display->setStatus(StatusProperty::Error, additional_topic_property_->getName(), QString("Error subscribing: ") + e.what());
177  }
178  }
179 
180  virtual void unsubscribe()
181  {
182  m_subscriber.unsubscribe();
183  }
184 
188  void incomingMessage(const typename MessageType::ConstPtr& msg)
189  {
190  if (!msg) {
191  return;
192  }
193 
194  ++m_messagesReceived;
195  m_display->setStatus(StatusProperty::Ok, additional_topic_property_->getName(), QString::number(m_messagesReceived) + " messages received");
196 
197  // Callback for further processing
198  m_messageCallback(msg);
199  }
200 
201 #if ROS_VERSION_MINIMUM(1, 14, 0)
202  tf2_ros::MessageFilter<MessageType>* tf_filter;
203 #else
204  tf::MessageFilter<MessageType>* tf_filter;
205 #endif
206 
207 private:
208  std::string m_topic;
209  bool m_enabled;
210  Display* m_display;
211  DisplayContext* m_context;
212  ros::NodeHandle m_updateNodeHandle;
213  message_filters::Subscriber<MessageType> m_subscriber;
214  uint32_t m_messagesReceived;
215 
216  const boost::function<void (boost::shared_ptr<const MessageType>)> m_messageCallback;
217 };
