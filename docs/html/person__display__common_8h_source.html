163  bool preprocessMessage(const typename MessageType::ConstPtr& msg)
164  {
165  // Here we call the rviz::FrameManager to get the transform from the
166  // fixed frame to the frame in the header of this Imu message. If
167  // it fails, we can't do anything else so we return.
168  if (!getContext()->getFrameManager()->getTransform(msg->header, m_framePosition, m_frameOrientation)) {
169  ROS_ERROR_THROTTLE(5.0, "Error transforming from frame '%s' into fixed frame!", msg->header.frame_id.c_str());
170  return false;
171  }
172 
173  m_frameOrientation.ToRotationMatrix(m_frameRotationMatrix);
174  return true;
175  }

211  // Builds velocity vector for a person from a twist message
212  Ogre::Vector3 getVelocityVector(const geometry_msgs::TwistWithCovariance& twist) {
213  const double zVelocityVariance = twist.covariance[2 * 6 + 2];
214  const double zVelocity = (isnan(zVelocityVariance) || isinf(zVelocityVariance)) ? 0.0 : twist.twist.linear.z;
215  return Ogre::Vector3(twist.twist.linear.x, twist.twist.linear.y, zVelocity);
216  }

219  bool hasValidOrientation(const geometry_msgs::PoseWithCovariance& pose)
220  {
221  // Check if quaternion has not been initialized, then it's invalid (all-zero elements)
222  if(pose.pose.orientation.x == 0 && pose.pose.orientation.y == 0 && pose.pose.orientation.z == 0 && pose.pose.orientation.w == 0) return false;
223 
224  // According to ROS conventions, orientation covariance is always fixed-frame
225  // so no transform necessary!
226  const double xRotVariance = pose.covariance[3 * 6 + 3];
227  const double yRotVariance = pose.covariance[4 * 6 + 4];
228  const double zRotVariance = pose.covariance[5 * 6 + 5];
229  // Using logical OR instead of AND here because the orientation is given as a quaternion, instead of independent
230  // RPY angles. We assume if at least one RPY angle is valid (according to its covariance), the other angles are set to a
231  // reasonable default (as part of the quaternion) if their variance is non-finite.
232  return xRotVariance < m_veryLargeVariance || yRotVariance < m_veryLargeVariance || zRotVariance < m_veryLargeVariance;
233  }

238  Ogre::Matrix3 covarianceXYZIntoTargetFrame(const geometry_msgs::PoseWithCovariance& pose) {
239  Ogre::Matrix3 xyzcov;
240  for(int row = 0; row < 3; row++) for(int col = 0; col < 3; col++) xyzcov[row][col] = pose.covariance[row*6 + col]; // 6 = dimension of ROS covariance matrix
241  if(!isfinite(xyzcov.Determinant())) ROS_WARN_STREAM("Covariance matrix supplied to covarianceXYZIntoTargetFrame() contains non-finite elements: " << xyzcov);
242  return m_frameRotationMatrix * xyzcov * m_frameRotationMatrix.Transpose(); // cov(AX + a) = A cov(X) A^T
243  }

246  void setPoseOrientation(Ogre::SceneNode* sceneNode, const geometry_msgs::PoseWithCovariance& pose, const Ogre::Matrix3& covXYZinTargetFrame, double personVisualHeight)
247  {
248  const geometry_msgs::Point& position = pose.pose.position;
249  const geometry_msgs::Quaternion& orientation = pose.pose.orientation;
250 
251  Ogre::Matrix4 transform(m_frameOrientation);
252  transform.setTrans(m_framePosition);
253 
254  Ogre::Vector3 originalPosition(position.x, position.y, position.z);
255  if(!isfinite(originalPosition.x) || !isfinite(originalPosition.y) || !isfinite(originalPosition.z)) {
256  ROS_WARN("Detected or tracked person has non-finite position! Something is wrong!");
257  return;
258  }
259 
260  Ogre::Vector3 positionInTargetFrame = transform * originalPosition;
261 
262  if(hasValidOrientation(pose)) {
263  Ogre::Quaternion detectionOrientation(orientation.w, orientation.x, orientation.y, orientation.z);
264  detectionOrientation.FromAngleAxis(detectionOrientation.getRoll(), Ogre::Vector3(0,0,1)); // only use yaw angle, ignore roll and pitch
265  sceneNode->setOrientation(m_frameOrientation * detectionOrientation);
266  }
267  else {
268  Ogre::Quaternion rotateTowardsCamera;
269  rotateTowardsCamera.FromAngleAxis(Ogre::Degree(180), Ogre::Vector3(0,0,1));
270  sceneNode->setOrientation(rotateTowardsCamera);
271  }
272 
273  const double zVariance = covXYZinTargetFrame[2][2];
274  bool useActualZPosition = m_commonProperties->use_actual_z_position->getBool() && isfinite(zVariance) && zVariance >= 0 && zVariance < m_veryLargeVariance;
275 
276  positionInTargetFrame.z = useActualZPosition ? positionInTargetFrame.z - personVisualHeight/2.0: 0.0;
277  positionInTargetFrame.z += m_commonProperties->z_offset->getFloat();
278 
279  sceneNode->setPosition(positionInTargetFrame);
280  }

283  Ogre::ColourValue getColorFromId(unsigned int object_id)
284  {
285  Ogre::ColourValue color;
286  const int colorScheme = m_commonProperties->color_transform->getOptionInt();
287  object_id += max(0, m_commonProperties->color_map_offset->getInt());
288 
289  if(colorScheme == COLORS_SRL || colorScheme == COLORS_SRL_ALTERNATIVE)
290  {
291  // SRL People Tracker colors
292  const size_t NUM_SRL_COLORS = 6, NUM_SRL_COLOR_SHADES = 4, NUM_SRL_TOTAL_COLORS = NUM_SRL_COLORS * NUM_SRL_COLOR_SHADES;
293  const unsigned int spencer_colors[NUM_SRL_TOTAL_COLORS] = {
294  0xC00000, 0xFF0000, 0xFF5050, 0xFFA0A0, // red
295  0x00C000, 0x00FF00, 0x50FF50, 0xA0FFA0, // green
296  0x0000C0, 0x0000FF, 0x5050FF, 0xA0A0FF, // blue
297  0xF20A86, 0xFF00FF, 0xFF50FF, 0xFFA0FF, // magenta
298  0x00C0C0, 0x00FFFF, 0x50FFFF, 0xA0FFFF, // cyan
299  0xF5A316, 0xFFFF00, 0xFFFF50, 0xFFFFA0 // yellow
300  };
301 
302  unsigned int rgb = 0;
303  const unsigned int tableId = object_id % NUM_SRL_TOTAL_COLORS;
304  if(m_commonProperties->color_transform->getOptionInt() == COLORS_SRL) {
305  // Colors in original order (first vary shade, then vary color)
306  rgb = spencer_colors[tableId];
307  }
308  else if(m_commonProperties->color_transform->getOptionInt() == COLORS_SRL_ALTERNATIVE) {
309  // Colors in alternative order (first vary color, then vary shade)
310  unsigned int shadeIndex = tableId / NUM_SRL_COLORS;
311  unsigned int colorIndex = tableId % NUM_SRL_COLORS;
312  rgb = spencer_colors[colorIndex * NUM_SRL_COLOR_SHADES + shadeIndex];
313  }
314 
315  float r = ((rgb >> 16) & 0xff) / 255.0f,
316  g = ((rgb >> 8) & 0xff) / 255.0f,
317  b = ((rgb >> 0) & 0xff) / 255.0f;
318 
319  color = Ogre::ColourValue(r, g, b, 1.0);
320  }
321  else if(colorScheme == COLORS_RAINBOW || colorScheme == COLORS_RAINBOW_BW)
322  {
323  const size_t NUM_COLOR = 10, NUM_BW = 4;
324  const unsigned int rainbow_colors[NUM_COLOR + NUM_BW] = {
325  0xaf1f90, 0x000846, 0x00468a, 0x00953d, 0xb2c908, 0xfcd22a, 0xffa800, 0xff4500, 0xe0000b, 0xb22222,
326  0xffffff, 0xb8b8b8, 0x555555, 0x000000
327  };
328 
329  color.setAsARGB(rainbow_colors[object_id % (colorScheme == COLORS_RAINBOW ? NUM_COLOR : (NUM_COLOR+NUM_BW))]);
330  }
331  else if(colorScheme == COLORS_FLAT)
332  {
333  const size_t NUM_COLOR = 10;
334  const unsigned int flat_colors[NUM_COLOR] = {
335  0x990033, 0xa477c4, 0x3498db, 0x1abc9c, 0x55e08f, 0xfff054, 0xef5523, 0xfe374a, 0xbaa891, 0xad5f43
336  };
337 
338  color.setAsARGB(flat_colors[object_id % NUM_COLOR]);
339  }
340  else if(colorScheme == COLORS_VINTAGE)
341  {
342  const size_t NUM_COLOR = 10;
343  const unsigned int vintage_colors[NUM_COLOR] = {
344  0xd05e56, 0x797d88, 0x448d7a, 0xa5d1cd, 0x88a764, 0xebe18c, 0xd8a027, 0xffcc66, 0xdc3f1c, 0xff9966
345  };
346 
347  color.setAsARGB(vintage_colors[object_id % NUM_COLOR]);
348  }
349  else
350  {
351  // Constant color for all tracks
352  color = m_commonProperties->constant_color->getOgreColor();
353  }
354 
355  color.a = 1.0f; // force full opacity
356  return color;
357  }

362  bool isIncluded = m_commonProperties->m_includedPersonIDs.find(personId) != m_commonProperties->m_includedPersonIDs.end();
