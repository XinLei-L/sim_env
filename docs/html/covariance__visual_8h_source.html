107  virtual void setMeanCovariance(const Ogre::Vector3& mean, const Ogre::Matrix3& cov)
108  {
109  int numberOfPoints;
110  double *xs, *ys;
111  double determinant = cov[0][0] * cov[1][1] - cov[1][0] * cov[0][1];
112 
113  m_line->clear();
114 
115  if (!std::isfinite(determinant))
116  {
117  ROS_WARN_STREAM_THROTTLE(
118  5.0, "Covariance matrix has non-finite values in ProbabilityEllipseCovarianceVisual::setMeanCovariance(): "
119  << cov);
120  return;
121  }
122 
123  if (std::abs(cov[0][1] - cov[1][0]) > 0.00001)
124  {
125  ROS_WARN_STREAM_THROTTLE(
126  5.0,
127  "Covariance matrix is not symmetric in ProbabilityEllipseCovarianceVisual::setMeanCovariance(): " << cov);
128  return;
129  }
130 
131  if (determinant > 0 && cov[0][0] > 0 /* positive definite? */ ||
132  std::abs(determinant - 0.00001) == 0.0 && (cov[0][0] > 0 || cov[1][1] > 0) /* positive semidefinite? */)
133  {
134  calc_prob_elli_99(mean.x, mean.y, cov[0][0], cov[1][1], cov[0][1], numberOfPoints, xs, ys);
135 
136  m_line->setMaxPointsPerLine(numberOfPoints);
137 
138  for (int i = 0; i < numberOfPoints; i++)
139  {
140  Ogre::Vector3 vertex(xs[i], ys[i], mean.z);
141  m_line->addPoint(vertex);
142  }
143  }
144  else
145  {
146  if (cov[0][0] != 0 || cov[0][1] != 0 || cov[0][2] != 0 || cov[1][0] != 0 || cov[1][1] != 0 || cov[1][2] != 0 ||
147  cov[2][0] != 0 || cov[2][1] != 0 || cov[2][2] != 0)
148  ROS_WARN_STREAM_THROTTLE(5.0,
149  "Covariance matrix is not positive (semi-)definite in "
150  "ProbabilityEllipseCovarianceVisual::setMeanCovariance(): "
151  << cov);
152  }
153  }

158  // Puts angle alpha into the interval [min..min+2*pi[
159  double set_angle_to_range(double alpha, double min)
160  {
161  while (alpha >= min + 2.0 * M_PI)
162  {
163  alpha -= 2.0 * M_PI;
164  }
165  while (alpha < min)
166  {
167  alpha += 2.0 * M_PI;
168  }
169  return alpha;
170  }

172  // Calculates the points on a rotated ellipse given by center xc, yc, half axes a, b and angle phi.
173  // Returns number of points np and points in Cart. coordinates
174  void calc_ellipse(double xc, double yc, double a, double b, double phi, int& np, double*& xvec, double*& yvec)
175  {
176  const int N_ELLI_POINTS = 40;
177  int i, offset;
178  double t, cr, sr, ca, sa, xi, yi, reso;
179  static double x[N_ELLI_POINTS + 1];
180  static double y[N_ELLI_POINTS + 1];
181  reso = 2 * M_PI / N_ELLI_POINTS;
182  offset = N_ELLI_POINTS / 2;
183  ca = cos(phi);
184  sa = sin(phi);
185  i = 0;
186  t = 0;
187  while (t < M_PI)
188  {
189  cr = cos(t);
190  sr = sin(t);
191  xi = a * cr * ca - b * sr * sa;
192  yi = a * cr * sa + b * sr * ca;
193  x[i] = xi + xc;
194  y[i] = yi + yc;
195  x[offset + i] = -xi + xc;
196  y[offset + i] = -yi + yc;
197  t = t + reso;
198  i++;
199  }
200  x[N_ELLI_POINTS] = x[0]; // Close contour
201  y[N_ELLI_POINTS] = y[0]; // Close contour
202  np = N_ELLI_POINTS + 1;
203  xvec = x;
204  yvec = y;
205  }

207  // Calculates the points on a 95%-iso-probability ellipse given by the bivarate RV with mean xc, yc
208  // and covariance matrix sxx, syy, sxy. Returns number of points np and points in Cart. coordinates
209  void calc_prob_elli_95(double xc, double yc, double sxx, double syy, double sxy, int& np, double*& x, double*& y)
210  {
211  double la, lb, a, b, phi;
212  la = (sxx + syy + sqrt((sxx - syy) * (sxx - syy) + 4 * sxy * sxy)) / 2;
213  lb = (sxx + syy - sqrt((sxx - syy) * (sxx - syy) + 4 * sxy * sxy)) / 2;
214  a = sqrt(5.991464 * la);
215  b = sqrt(5.991464 * lb);
216  phi = set_angle_to_range(atan2(2 * sxy, sxx - syy) / 2, 0);
217  calc_ellipse(xc, yc, a, b, phi, np, x, y);
218  }
219 
220  // Calculates the points on a 99%-iso-probability ellipse given by the bivarate RV with mean xc, yc
221  // and covariance matrix sxx, syy, sxy. Returns number of points np and points in Cart. coordinates
222  void calc_prob_elli_99(double xc, double yc, double sxx, double syy, double sxy, int& np, double*& x, double*& y)
223  {
224  double la, lb, a, b, phi;
225  la = (sxx + syy + sqrt((sxx - syy) * (sxx - syy) + 4 * sxy * sxy)) / 2;
226  lb = (sxx + syy - sqrt((sxx - syy) * (sxx - syy) + 4 * sxy * sxy)) / 2;
227  a = sqrt(9.210340 * la);
228  b = sqrt(9.210340 * lb);
229  phi = set_angle_to_range(atan2(2 * sxy, sxx - syy) / 2, 0);
230  calc_ellipse(xc, yc, a, b, phi, np, x, y);
231  }
