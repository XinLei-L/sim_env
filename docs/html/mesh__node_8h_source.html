51  class MeshNode : public Ogre::FrameListener {
52  public:
53  MeshNode(rviz::DisplayContext* displayContext, Ogre::SceneNode* parentNode, const std::string& meshResource, Ogre::Vector3 position = Ogre::Vector3::ZERO)
54  : m_sceneManager(displayContext->getSceneManager()), m_displayContext(displayContext), m_meshResource(meshResource)
55  {
56  m_cameraFacing = false;
57  m_sceneNode = parentNode->createChildSceneNode();
58  m_sceneNode->setVisible(false);
59 
60  // Load mesh
61  assert(!rviz::loadMeshFromResource(meshResource).isNull());
62 
63  // Create scene entity
64  std::stringstream ss;
65  static int counter = 0;
66  ss << "gender_symbol_" << counter++;
67  std::string id = ss.str();
68 
69  m_entity = m_sceneManager->createEntity(id, meshResource);
70  m_sceneNode->attachObject(m_entity);
71 
72  // set up material
73  ss << "Material";
74  // Ogre::MaterialPtr default_material = Ogre::MaterialManager::getSingleton().create( ss.str(), "rviz" );
75  Ogre::MaterialPtr default_material = Ogre::MaterialManager::getSingleton().create(ss.str(), Ogre::ResourceGroupManager::DEFAULT_RESOURCE_GROUP_NAME);
76  default_material->setReceiveShadows(false);
77  default_material->getTechnique(0)->setLightingEnabled(true);
78  default_material->getTechnique(0)->setAmbient( 0.5, 0.5, 0.5 );
79  m_materials.insert( default_material );
80  m_entity->setMaterial( default_material );
81 
82  setPosition(position);
83  setVisible(true);
84 
85  // For camera-facing meshes
86  Ogre::Root::getSingleton().addFrameListener(this);
87  }
88 
89  virtual ~MeshNode() {
90  Ogre::Root::getSingleton().removeFrameListener(this);
91  m_sceneManager->destroyEntity(m_entity);
92 
93  // destroy all the materials we've created
94  std::set<Ogre::MaterialPtr>::iterator it;
95  for(it = m_materials.begin(); it != m_materials.end(); it++ )
96  {
97  Ogre::MaterialPtr material = *it;
98  if (!material.isNull())
99  {
100  material->unload();
101  Ogre::MaterialManager::getSingleton().remove(material->getName());
102  }
103  }
104  m_materials.clear();
105  m_sceneManager->destroySceneNode(m_sceneNode->getName());
106  }

128  void setColor(const Ogre::ColourValue& c) {
129  Ogre::SceneBlendType blending;
130  bool depth_write;
131 
132  if ( c.a < 0.9998 )
133  {
134  blending = Ogre::SBT_TRANSPARENT_ALPHA;
135  depth_write = false;
136  }
137  else
138  {
139  blending = Ogre::SBT_REPLACE;
140  depth_write = true;
141  }
142 
143  std::set<Ogre::MaterialPtr>::iterator it;
144  for(it = m_materials.begin(); it != m_materials.end(); it++)
145  {
146  Ogre::Technique* technique = (*it)->getTechnique( 0 );
147 
148  technique->setAmbient( c.r*0.5, c.g*0.5, c.b*0.5 );
149  technique->setDiffuse( c.r, c.g, c.b, c.a );
150  technique->setSceneBlending( blending );
151  technique->setDepthWriteEnabled( depth_write );
152  technique->setLightingEnabled( true );
153  }
154  }

160  // We are using this FrameListener callback to orient the mesh towards the camera.
161  // Using a SceneManager::Listener and its preUpdateSceneGraph(SceneManager, Camera) method doesn't work because
162  // it is apparently never invoked by the Rviz render system.
163  virtual bool frameStarted(const Ogre::FrameEvent &evt)
164  {
165  Ogre::Quaternion cameraQuat;
166  if(m_cameraFacing) {
167  // Align with camera view direction
168  // FIXME: The following way of retrieving the camera and its position is a bit hacky, don't try this at home!
169  rviz::VisualizationManager* visualizationManager = dynamic_cast<rviz::VisualizationManager*>(m_displayContext);
170  assert(visualizationManager != NULL);
171  cameraQuat = visualizationManager->getRenderPanel()->getCamera()->getOrientation();
172  }
173  m_sceneNode->setOrientation(cameraQuat * m_orientation);
174  return true;
175  }
