58  class PersonVisual {
59  public:
60  PersonVisual(const PersonVisualDefaultArgs& args) :
61  m_sceneManager(args.sceneManager),
62  m_correctOrientation( Ogre::Degree(90), Ogre::Vector3(1,0,0) )
63  {
64  m_parentSceneNode = args.parentNode;
65  m_sceneNode = args.parentNode->createChildSceneNode();
66 
67  Ogre::Vector3 scale(1,1,1);
68  m_sceneNode->setScale(m_correctOrientation * scale);
69  }
70 
71  virtual ~PersonVisual() {
72  m_sceneManager->destroySceneNode(m_sceneNode->getName());
73  };
74 
75  void setPosition(const Ogre::Vector3& position) {
76  m_sceneNode->setPosition(position);
77  }
78 
79  const Ogre::Vector3& getPosition() const {
80  return m_sceneNode->getPosition();
81  }
82 
83  void setOrientation(const Ogre::Quaternion& orientation) {
84  m_sceneNode->setOrientation(orientation * m_correctOrientation);
85  }
86 
87  const Ogre::Quaternion& getOrientation() const {
88  return m_sceneNode->getOrientation();
89  }
90 
91  virtual void setScalingFactor(double scalingFactor) {
92  m_sceneNode->setScale(scalingFactor, scalingFactor, scalingFactor);
93  }
94 
95  void setVisible(bool visible) {
96  m_sceneNode->setVisible(visible, true);
97  }
98 
99  Ogre::SceneNode* getParentSceneNode() {
100  return m_parentSceneNode;
101  }
102 
103  virtual void update(float deltaTime) {}
104  virtual void setColor(const Ogre::ColourValue& c) = 0;
105  virtual double getHeight() = 0;
106 
107  protected:
108  Ogre::SceneManager* m_sceneManager;
109  Ogre::SceneNode *m_sceneNode, *m_parentSceneNode;
110  Ogre::Quaternion m_correctOrientation;
111  };

115  class CylinderPersonVisual : public PersonVisual {
116  public:
117  CylinderPersonVisual(const PersonVisualDefaultArgs& args) : PersonVisual(args)
118  {
119  m_bodyShape = new rviz::Shape(rviz::Shape::Cylinder, args.sceneManager, m_sceneNode);
120  m_headShape = new rviz::Shape(rviz::Shape::Sphere, args.sceneManager, m_sceneNode);
121 
122  const float headDiameter = 0.4;
123  const float totalHeight = getHeight();
124  const float cylinderHeight = totalHeight - headDiameter;
125 
126  m_bodyShape->setScale(Ogre::Vector3(headDiameter, headDiameter, cylinderHeight));
127  m_headShape->setScale(Ogre::Vector3(headDiameter, headDiameter, headDiameter));
128 
129  m_bodyShape->setPosition(Ogre::Vector3(0, 0, cylinderHeight / 2 - totalHeight / 2));
130  m_headShape->setPosition(Ogre::Vector3(0, 0, totalHeight / 2 - headDiameter / 2 ));
131  }
132 
133  virtual ~CylinderPersonVisual() {
134  delete m_bodyShape;
135  delete m_headShape;
136  }
137 
138  virtual void setColor(const Ogre::ColourValue& c) {
139  m_bodyShape->setColor(c);
140  m_headShape->setColor(c);
141  }
142 
143  virtual double getHeight() {
144  return 1.75;
145  }
146 
147  private:
148  rviz::Shape *m_bodyShape, *m_headShape;
149  };

153  class BoundingBoxPersonVisual : public PersonVisual, public HasLineWidth {
154  public:
155  BoundingBoxPersonVisual(const PersonVisualDefaultArgs& args, double height = 1.75, double width = 0.6, double scalingFactor = 1.0) : PersonVisual(args)
156  {
157  m_width = width; m_height = height; m_scalingFactor = scalingFactor; m_lineWidth = 0.03;
158  m_wireframe = NULL;
159  generateWireframe();
160  }
161 
162  virtual ~BoundingBoxPersonVisual() {
163  delete m_wireframe;
164  }

166  virtual void setColor(const Ogre::ColourValue& c) {
167  m_wireframe->setColor(c.r, c.g, c.b, c.a);
168  }
169 
170  virtual double getHeight() {
171  return m_height;
172  }
173 
174  virtual void setLineWidth(double lineWidth) {
175  m_wireframe->setLineWidth(lineWidth);
176  }
177 
178  /*
179  virtual void setScalingFactor(double scalingFactor) {
180  if(scalingFactor != m_scalingFactor) {
181  m_scalingFactor = scalingFactor;
182  generateWireframe();
183  }
184  }
185  */
186 
187  protected:
188  virtual void generateWireframe() {
189  delete m_wireframe;
190  m_wireframe = new rviz::BillboardLine(m_sceneManager, m_sceneNode);
191 
192  m_wireframe->setLineWidth(m_lineWidth);
193  m_wireframe->setMaxPointsPerLine(2);
194  m_wireframe->setNumLines(12);
195 
196  double w = m_width * m_scalingFactor, h = m_height * m_scalingFactor;
197  Ogre::Vector3 bottomLeft(0, -w, 0), bottomRight(0, 0, 0), topLeft(0, -w, h), topRight(0, 0, h);
198  Ogre::Vector3 rear(w, 0, 0);
199 
200  // Front quad
201  m_wireframe->addPoint(bottomLeft); m_wireframe->addPoint(bottomRight);
202  m_wireframe->newLine(); m_wireframe->addPoint(bottomRight); m_wireframe->addPoint(topRight);
203  m_wireframe->newLine(); m_wireframe->addPoint(topRight); m_wireframe->addPoint(topLeft);
204  m_wireframe->newLine(); m_wireframe->addPoint(topLeft); m_wireframe->addPoint(bottomLeft);
205 
206  // Rear quad
207  m_wireframe->newLine(); m_wireframe->addPoint(bottomLeft + rear); m_wireframe->addPoint(bottomRight + rear);
208  m_wireframe->newLine(); m_wireframe->addPoint(bottomRight + rear); m_wireframe->addPoint(topRight + rear);
209  m_wireframe->newLine(); m_wireframe->addPoint(topRight + rear); m_wireframe->addPoint(topLeft + rear);
210  m_wireframe->newLine(); m_wireframe->addPoint(topLeft + rear); m_wireframe->addPoint(bottomLeft + rear);
211 
212  // Four connecting lines between front and rear
213  m_wireframe->newLine(); m_wireframe->addPoint(bottomLeft); m_wireframe->addPoint(bottomLeft + rear);
214  m_wireframe->newLine(); m_wireframe->addPoint(bottomRight); m_wireframe->addPoint(bottomRight + rear);
215  m_wireframe->newLine(); m_wireframe->addPoint(topRight); m_wireframe->addPoint(topRight + rear);
216  m_wireframe->newLine(); m_wireframe->addPoint(topLeft); m_wireframe->addPoint(topLeft + rear);
217 
218  m_wireframe->setPosition(Ogre::Vector3(-w/2, w/2, -h/2));
219  }
220 
221  private:
222  rviz::BillboardLine *m_wireframe;
223  double m_width, m_height, m_scalingFactor, m_lineWidth;
224  };
