332  void _nnSearchRecursive(const PointT& query, const KDNode* node, int* guess, double* min_dist) const
333  {
334  if (node == nullptr)
335  return;
336

347  const int dir = query[axis] < train[axis] ? 0 : 1;
348  _nnSearchRecursive(query, node->next[dir], guess, min_dist);
349

351  // in the other side of axis, therefore another direction should be searched
352  const double diff = fabs(query[axis] - train[axis]);
353  if (diff < *min_dist)
354  _nnSearchRecursive(query, node->next[!dir], guess, min_dist);
355  }
